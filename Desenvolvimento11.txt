def calculadora(num1, num2, operacao):
	if operacao == 1:
		return num1 + num2
	elif operacao == 2:
		return num1 - num2
	elif operacao == 3: 
		return num1 * num2
	elif operacao == 4:
	    return num1 / num2

while calculadora(num1, num2, operacao):
	num1 = float(input("Digite o primeiro número: "))
	num2 = float(input("Digite o segundo número: "))
	operacao = int(input("Digite a operação 1 para somar. 2 para subtrair. 3 para multiplicar. 4 para dividir): "))

	resultado = calculadora(num1, num2, operacao)
	print("Resultado:", resultado)
	if (operacao == 0):
		break
